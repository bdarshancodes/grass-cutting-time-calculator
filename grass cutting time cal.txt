#include<stdio.h>
void main(){
    int hl,yl,hw,yw,g,house,yard;
    float final;
    printf("enter the lenght and width of house\n");
    scanf("%d%d",&hl,&hw);
    printf("enter the lenght and width of yard\n");
    scanf("%d%d",&yl,&yw);

}
